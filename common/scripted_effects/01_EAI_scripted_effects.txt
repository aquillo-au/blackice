queue_INF = { # queue INF anywhere possible
	if = {
		limit = { any_state = { available_INF = yes } }
		if = {
			limit = { any_state = { available_INF = yes available_state_0 = yes is_in_home_area = yes } } random_state = { limit = { available_INF = yes available_state_0 = yes is_in_home_area = yes } add_INF = yes }
			else = {
				if = {
					limit = { any_state = { available_INF = yes is_in_home_area = yes } } random_state = { limit = { available_INF = yes is_in_home_area = yes } add_INF = yes }
					else = {
						if = {
							limit = { any_state = { available_INF = yes available_state_0 = yes } } random_state = { limit = { available_INF = yes available_state_0 = yes } add_INF = yes }
							else = {
								random_state = { limit = { available_INF = yes } add_INF = yes }
							}
						}
					}
				}
			}
		}
		else = {
			queue_RADAR = yes
		}
	}
}

queue_INF_2 = { # see if there is a good place for infrastructure, else queue factories
	if = {
		limit = {
			ROOT = { num_of_civilian_factories > 9 }
			available_INF = yes
			is_in_home_area = yes
		}

		add_INF = yes

		else = {
			if = {
				limit = { available_state_0 = yes ROOT = { num_of_civilian_factories < 10 } }

				add_CIC = yes

				else = {
					if = {
						limit = { available_state_0 = yes }

						add_MIC = yes

						else = {
							random_state = {
								limit = {
									available_INF = yes
									is_in_home_area = yes
								}
								add_INF = yes
							}
						}
					}
				}
			}
		}
	}
}

queue_INF_supply = { # for building INF to increase supply
	if = {
		limit = {
			any_state = {
				available_INF = yes
				free_building_slots = { building = infrastructure size > 5 include_locked = no }
				ROOT = { divisions_in_state = { state = PREV size > 5 } }
				any_neighbor_state = { CONTROLLER = { has_war_with = ROOT strength_ratio = { tag = ROOT ratio > 0.5 } } }
			}
		}

			limit = {
				available_INF = yes

				limit = {
					any_state = {
						available_INF = yes

						ROOT = { divisions_in_state = { state = PREV size > 5 } }

				}

					limit = {
						available_INF = yes

			}
		}
	}
}

									limit = { any_state = { available_RADAR = yes is_in_home_area = yes NOT = { any_state = { OR = { has_state_flag = radar_site radar_station > 0 } CONTROLLER = { OR = { tag = ROOT is_in_faction_with = ROOT } } distance_to = { target = PREV value < 800 } } } } }

queue_RADAR_random = { # queue radar in a good random location
	if = {
		limit = {
			any_state = {
				available_RADAR = yes
				NOT = { any_state = { OR = { has_state_flag = radar_site radar_station > 0 } CONTROLLER = { OR = { tag = ROOT is_in_faction_with = ROOT } } distance_to = { target = PREV value < 800 } } }
				NOT = { any_neighbor_state = { OR = { has_state_flag = radar_site radar_station > 0 } CONTROLLER = { OR = { tag = ROOT is_in_faction_with = ROOT } } } }
				NOT = { any_neighbor_state = { CONTROLLER = { NOT = { tag = ROOT is_in_faction_with = ROOT } } } }
			}
		}
		random_state = {
			limit = {
				available_RADAR = yes
				NOT = { any_state = { OR = { has_state_flag = radar_site radar_station > 0 } CONTROLLER = { OR = { tag = ROOT is_in_faction_with = ROOT } } distance_to = { target = PREV value < 800 } } }
				NOT = { any_neighbor_state = { OR = { has_state_flag = radar_site radar_station > 0 } CONTROLLER = { OR = { tag = ROOT is_in_faction_with = ROOT } } } }
				NOT = { any_neighbor_state = { CONTROLLER = { NOT = { tag = ROOT is_in_faction_with = ROOT } } } }
			}
			add_RADAR = yes
		}
	}
}

queue_AA = { # build AA on synth factories if facing strategic bombers
	if = {
		limit = {
			any_enemy_country = { has_deployed_air_force_size = { size > 200 type = strategic_bomber } }

			is_major = yes
			any_state = {
				is_controlled_by = ROOT
				free_building_slots = { building = anti_air_building size > 0 include_locked = no }
				OR = { synthetic_refinery > 0 radar_station > 2 }
			}
		}
		random_state = {
			limit = {
				is_controlled_by = ROOT
				free_building_slots = { building = anti_air_building size > 0 include_locked = no }
				OR = { synthetic_refinery > 0 radar_station > 2 }
			}
			add_building_construction = { type = anti_air_building level = 1 instant_build = no }
		}
	}
}
