
stop_cv_air = {
	enable = {
		has_navy_size = { 
			size < 1
			type = carrier
		}
	}
	abort = {
		has_navy_size = {
			size > 0
			type = carrier
		}
	}
	
	ai_strategy = { type = equipment_production_factor id = interceptor 				value = -100		}	
	ai_strategy = { type = unit_ratio id = interceptor 							value = -100		}
	ai_strategy = { type = build_army id = interceptor 							value = -100 	}	
}

		}

		id = relation
		value = -9999
	}
}

save_admiral_pp_for_later = {
	enable = {
		has_war = no
		threat < 0.3
		has_political_power < 200		
	}
	abort = {
		OR = {
			has_war = yes	
			threat > 0.3
			has_political_power > 200
		}
	}

	ai_strategy = { # dont burn PP on admirals (can still get free ones)
		type = pp_spend_priority
		id = admiral
		value = -100
	}
}
