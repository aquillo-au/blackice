stay_in_allies = {
    enable = {
		tag = FRA
		has_country_flag = in_allies
		is_in_faction_with = ENG
		NOT = { is_faction_leader = yes }
        ENG = { is_faction_leader = yes }
    }
    abort = {
		ENG = { 
			is_faction_leader = no
			NOT = { has_country_flag = in_allies }
		}
		democratic_govt = no
    }
 
    ai_strategy = { type = alliance id = "ENG"     		        value = 300000     }
}
