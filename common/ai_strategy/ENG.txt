	enable = {
		tag = ENG
		democratic_govt = yes
		is_historical_focus_on = yes
		has_war_with = ITA
		NOT = { 
			ITA = { has_capitulated = yes }
			ITA = { exists = no }
			ITA = { is_puppet = yes }
			is_in_faction_with = ITA
		}
		ITA = { has_government = fascism }
	}
	abort = {
		OR = { 
			ITA = { has_capitulated = yes }
			ITA = { exists = no }
			ITA = { is_puppet = yes }
			is_in_faction_with = ITA
			NOT = { ITA = { has_government = fascism } }
			
			NOT = { has_war_with = ITA }
			NOT = { democratic_govt = yes }
		}
	}
	
	ai_strategy = {
		type = conquer
		id = "ITA"
		value = 1000
	}
}
