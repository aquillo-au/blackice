CAN_hunt_Italy = {
	enable = {
		original_tag = CAN	
		NOT = { has_capitulated = yes }
		any_owned_state = { OR = { region = 245 region = 128 region = 126 } controller = { has_war_with = GER }}
		}
		abort = {
			OR = {
				has_capitulated = yes	
				NOT = { any_owned_state = { OR = { region = 245 region = 128 region = 126 } controller = { has_war_with = GER }}}
				}
			}
			ai_strategy = {
				type = invade
				id = "ITA"
				value = 500
			}

				type = antagonize
				id = "ITA"
				value = 500
			}
			ai_strategy = {
				type = conquer
				id = "ITA"
				value = 500
			}
		}
