add_namespace = china2


	#Remove debuffs if CHI wins civil war
country_event = {
	id = china2.215
	title = china2.215.t
	desc = china2.215.d
	picture = GFX_report_event_chinese_soldiers

	fire_only_once = yes

	trigger = {
		tag = CHI
		NOT = {
			country_exists = PRC
		}
	}

	mean_time_to_happen = {
		days = 10
	}

	option = {
		name = china2.215.a
		
			remove_ideas = CHI_communist_uprisings
			remove_ideas = prc_influence
			remove_ideas = prc_propaganda
					
			add_popularity = {
				ideology = communism
				popularity = -1
		}
	}



	}

# China turns communist / Communists break off from China
country_event = {
	id = china2.666
	title = china2.666.t
	desc = china2.666.d
	picture = GFX_report_event_generic_sign_treaty1

	mean_time_to_happen = {
		days = 2
	}

	trigger = {
	
	OR = {
	original_tag = XSM
	original_tag = SHX
	original_tag = GXC
	original_tag = YUN
	}
		has_government = communism
		PRC = {
			exists = yes
			has_government = communism
		}
	}

	option = {
		name = china2.666.a
		hidden_effect = { PRC = { country_event = { id = china2.667 } } }
		effect_tooltip = {
			PRC = {
				annex_country = { target = ROOT transfer_troops = yes }
			}
		}
		custom_effect_tooltip = GAME_OVER_TT
	}
}

# PRC annexes communist revolters from China
country_event = {
	id = china2.667
	title = china2.667.t
	desc = china2.667.d
	picture = GFX_report_event_generic_sign_treaty1

	is_triggered_only = yes

	option = {
		name = china2.667.a
		if = {
			limit = {
				FROM = {
					has_civil_war = yes
				}
			}
			random_other_country = {
				limit = {
						OR = {
	original_tag = XSM
	original_tag = SHX
	original_tag = GXC
	original_tag = YUN
	}
					has_war_with = FROM
				}
				declare_war_on = { target = ROOT type = annex_everything }
			}
		}
		PRC = {
			annex_country = { target = FROM transfer_troops = yes }
		}
	}
}
