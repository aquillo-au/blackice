########################### Uniform Factories AI #################
country_event = {
	id = buildings.13
	title = buildings.13.t
	desc = buildings.13.d
	
	picture = GFX_report_event_german_speech

	mean_time_to_happen = {
        days = 2
    }

	trigger = {
		is_ai = yes
		date > 1936.1.1
		OR = {
			has_tech = uniform_industry
			has_tech = dispersed_uniform_industry
		}
		NOT = { has_country_flag = uniforms_factory_one }
	}
	
	immediate = {
		set_country_flag = uniforms_factory_one
	}
	
	option = {
		name = buildings.13.a
		random_owned_state = {
			limit = { 
				free_building_slots = {
					building = uniforms_factory
					size > 0
					include_locked = yes
				}
			}
			add_extra_state_shared_building_slots = 2
			add_building_construction = {
				type = uniforms_factory
				level = 1
				instant_build = yes
			}
		}
	}
}
