country_event = {
	id = angry_level.1
	title = angry_level.1.t
	desc = angry_level.1.desc
	hidden = yes
	
	trigger = {
		date > 1936.1.1
		tag = GER
		has_global_flag = allied_anger_1
		NOT = { has_country_flag = allied_anger_1a }
	}
	
	mean_time_to_happen = {
		days = 2
	}
	
	immediate = { 
		set_country_flag = allied_anger_1a
		clr_country_flag = allied_anger_2a

		clr_country_flag = allied_anger_9a
		clr_country_flag = allied_anger_10a
	}

		name = angry_level.1.a
		
		add_ideas = allied_anger_1

	}
}
