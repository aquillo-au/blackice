	picture = GFX_report_event_soviet_soldiers2

	trigger = {
		tag = SOV
		has_war_with = GER
		is_ai = yes
		#GER = {is_ai = no}
		is_puppet = no
		exists = yes 
		NOT = { has_idea = BalticAttack_I }
		NOT = { has_idea = BalticAttack_II }
		NOT = { has_idea = BalticAttack_III }
		NOT = { has_idea = CounterAttack_II }
		NOT = { has_idea = CounterAttack_III }
		NOT = { has_idea = CounterAttack_IV }
		or = {
		NOT = { controls_state = 97 }
		NOT = { controls_state = 91 }
		NOT = { controls_state = 94 }
		NOT = { controls_state = 93 }
		
		}
		#has_country_flag = forthemotherland
	}
	
	mean_time_to_happen = {
		days = 1
	}
	
	option = { 
		name = soviet.312.a
		hidden_effect = {
			
			add_timed_idea = {
				idea = CounterAttack_I
				days = 7
			}
	}

}
