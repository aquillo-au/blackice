# TIB Receives Ultimatum
country_event = {
	id = comchina.3
	title = comchina.3.t
	desc = comchina.3.d
	picture = GFX_report_event_albanian_king_zog
	
	is_triggered_only = yes
	
	option = { # Yield
		name = comchina.3.a
		ai_chance = {
			base = 40
			modifier = {
				factor = 0				
				any_other_country = {
					is_major = yes
					NOT = { tag = PRC }
					OR = {
						has_guaranteed = TIB
						is_in_faction_with = TIB
					}
				}
			}
		}
		PRC = {
			country_event = { days = 1 id = comchina.1 }
		}
		custom_effect_tooltip = GAME_OVER_TT
	}
	option = { # Resist
		name = comchina.3.b
		ai_chance = {
			factor = 20 
			modifier = {
				factor = 0
				OR = {
					AND = {
						is_in_faction_with = PRC
						PRC = { is_faction_leader = yes }
					}
					is_historical_focus_on = yes
					AND = {
						is_in_faction = no 
						NOT = {
							any_other_country = {
								NOT = { tag = PRC }
								has_guaranteed = TIB
							} 
						}
					}
				}
			}
		}
		PRC = {
			effect_tooltip = {
				create_wargoal = {
					type = annex_everything
					target = TIB
				}
			}
			country_event = { days = 1 id = comchina.2 }
		}
	}
}
